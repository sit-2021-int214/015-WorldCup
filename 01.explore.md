### 1.) คู่ไหนที่ยิงประตูครึ่งหลังรวมกันเกิน 4 ประตู
```{R}

halfGoals <- worldcup %>% 
  select(Home.Team.Name,Half.time.Home.Goals,Away.Team.Name,Half.time.Away.Goals) %>% 
  rename(Home.Team.Goals = Half.time.Home.Goals,Away.Team.Goals = Half.time.Away.Goals) %>% 
  filter(Away.Team.Goals + Home.Team.Goals > 4) 
halfGoals <- halfGoals %>% 
  mutate(Total.Goals = Home.Team.Goals + Away.Team.Goals) 
View(halfGoals)

```
